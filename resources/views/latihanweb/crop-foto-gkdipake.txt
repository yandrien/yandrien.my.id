    ctx.drawImage(img, 0, 0, cropCanvas.width, cropCanvas.height);
    

  // Menghitung ukuran crop default 4x6 cm

let isDragging = false;
let mouseX, mouseY;

const handleStart = (e) => {
  e.preventDefault();
  const rect = cropCanvas.getBoundingClientRect();
  const x = (e.touches ? e.touches[0].clientX : e.clientX) - rect.left;
  const y = (e.touches ? e.touches[0].clientY : e.clientY) - rect.top;
  if (x >= cropX * scale && x <= (cropX + cropW) * scale && y >= cropY * scale && y <= (cropY + cropH) * scale) {
    isDragging = true;
    mouseX = x;
    mouseY = y;
  }
};


const handleMove = (e) => {
  e.preventDefault();
  if (isDragging) {
    const rect = cropCanvas.getBoundingClientRect();
    const x = (e.touches ? e.touches[0].clientX : e.clientX) - rect.left;
    const y = (e.touches ? e.touches[0].clientY : e.clientY) - rect.top;
    cropX += (x - mouseX) / scale;
    cropY += (y - mouseY) / scale;
    mouseX = x;
    mouseY = y;
    ctx.clearRect(0, 0, cropCanvas.width * scale, cropCanvas.height * scale);
    ctx.drawImage(img, 0, 0, cropCanvas.width * scale, cropCanvas.height * scale);
    // Membuat area luar crop menjadi agak gelap
    ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
    ctx.fillRect(0, 0, cropCanvas.width * scale, cropY * scale); // atas
    ctx.fillRect(0, cropY * scale + cropH, cropCanvas.width * scale, cropCanvas.height * scale - (cropY * scale + cropH)); // bawah
    ctx.fillRect(0, cropY * scale, cropX * scale, cropH); // kiri
    ctx.fillRect(cropX * scale + cropW, cropY * scale, cropCanvas.width * scale - (cropX * scale + cropW), cropH); // kanan
    ctx.strokeStyle = 'red';
    ctx.lineWidth = 2;
    //buat kotak merah
    ctx.strokeRect(cropX * scale, cropY * scale, cropW, cropH);
  }
};



const handleEnd = () => {
  isDragging = false;
};

  cropCanvas.addEventListener('mousemove', handleMove);
  cropCanvas.addEventListener('touchend', handleEnd);
  cropCanvas.addEventListener('mouseup', handleEnd);

cropOk.addEventListener('click', () => {
  const croppedCanvas = document.createElement('canvas');
  croppedCanvas.width = 80;
  croppedCanvas.height = 100;
  const croppedCtx = croppedCanvas.getContext('2d');
  const ratio = cropW / cropH;
  if (ratio > 0.8) {
    croppedCtx.drawImage(img, cropX * (img.width / cropCanvas.width), cropY * (img.height / cropCanvas.height), cropW * (img.width / cropCanvas.width), cropW * (img.height / cropCanvas.height) / ratio, 0, 0, 80, 80 / ratio);
  } else {
    croppedCtx.drawImage(img, cropX * (img.width / cropCanvas.width), cropY * (img.height / cropCanvas.height), cropH * (img.width / cropCanvas.width) * ratio, cropH * (img.height / cropCanvas.height), 0, 0, 80 * ratio, 100);
  }
  pic.src = croppedCanvas.toDataURL();
  
  // simpan blob ke variabel croppedFile
  cropCanvas.toBlob((blob) => {
  croppedFile = blob;
  });
  
  //kirim hasil crop saat disubmit ke server
  document.getElementById('formd').addEventListener('submit', (e) => {
      const formData = new FormData(e.target);
      formData.append('foto-crop', croppedFile);
  });
  
  cropModal.style.display = 'none';
  
});

  ctx.fillRect(0, cropY, cropX, cropH);
